Importante:

para agregar traducción local a nuestros pipes, debemos agregar lo siguiente (omitiendo las comillas dobles) a nuestro módulo: 

    {{ "import { LOCALE_ID, NgModule } from '@angular/core';" }}
    {{ "import { registerLocaleData } from '@angular/common';" }}

    // importar locales
    import localeEsAr from '@angular/common/locales/es-AR';

    providers: {{"[{ provide: LOCALE_ID, useValue: 'es-Ar' }]" }},

Para agregar el efecto ripple en el botón del fieldset de PrimeNG debemos agregar lo siguiente en el componente donde se está usando o incluso en el app.component.ts:

    {{ "import { PrimeNGConfig } from 'primeng/api';" }}

    constructor(private primengConfig: PrimeNGConfig)

    ngOnInit() {{ "{
      this.primengConfig.ripple = true;
    }" }}
